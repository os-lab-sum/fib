#include 

int main() {
    int n;
    unsigned long long int t1 = 0, t2 = 1;
    unsigned long long int nxt;
    printf("Enter the number of terms for the Fibonacci series: ");
    scanf("%d", &n);
    if (n <= 0) {
        printf("Please enter a positive number of terms.\n");
        return 1; 
    }
    printf("Fibonacci Series for %d terms:\n", n);
    for (int i = 1; i <= n; ++i) {
        if (i == 1) {
            printf("%llu", t1);
            continue;
        }
        if (i == 2) {
            printf(", %llu", t2);
            continue;
        }
        nxt = t1 + t2;
        t1 = t2;
        t2 = nxt;
        printf(", %llu", nxt);
    }
    printf("\n");
    return 0; 
}
